# Combine the data and calculate the difference in row numbers
combined_data = []
for i in range(1, 51):
    row_data21 = data21[i]
    name21 = row_data21[1]
    row_data23 = next((row for row in data23 if row[0] == name21), None)
    if row_data23:
        row_number_difference = i - data23.index(row_data23)
    else:
        row_data23 = [""] * 3  # Fill with empty values if no match is found in data23
        row_number_difference = ""

    combined_row = row_data21 + row_data23 + [row_number_difference]
    combined_data.append(combined_row)

# Display the combined table
headers = ["Rank21", "Name21", "Followers21", "Name23", "Followers23", "Country", "Row Difference"]
print(tabulate(combined_data, headers=headers))
