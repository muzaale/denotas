def create_table(names):
    header = ["Number", "Name"]
    table = []
    table.append(header)
    for idx, name in enumerate(names, start=1):
        full_name = " ".join(name)
        row = [idx, full_name]
        table.append(row)
    
    # Printing the table
    for row in table:
        print(f"{row[0]:<10} {row[1]:<30}")

# Call the function
create_table(names)
